e je  L2 /* goto L2 */, é o caso do while, mas em assembly é condicionado inverso (se o if estiver correto e cair para o L2,

2)Ao mudar o código em C para if (*pc != '{' && *pc !='}') -> printf ("%c", *pc++); e o código de assembly para 
  cmpb $'{', (%r12)
  je Final_while
  cmpb $'}', (%r12)
  je Final_while

Final_while:
  addq  $1, %r12  /* r12 += 1; */
  jmp  L1         /* goto L1; */
  
fazendo um curto circuito para caso caia na primeira validação, não seja necessário verificar a segunda condição,
tem-se o printf Alo galera!

um registrador para guardar o primeiro endereço do array (%r12), um registrador para armazenar o valor de um elemento (%rcx) e,
por último, uma registrador para fazer a soma (%r13d). Dessa forma, a partir desse passo a passo:

  cmpl $4, %ebx
  jge L2
  movl %ebx, %edx
  imull $4, %edx
  movq %r12, %rcx
  addq %rdx, %rcx
  addl (%rcx), %r13d
